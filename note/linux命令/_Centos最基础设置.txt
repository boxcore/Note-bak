

---------------------------------------------
一、设置启动方式
---------------------------------------------
vi /etc/inittab 

设置id:5:initdefault 为 id:3:initdefault


---------------------------------------------
二、设置语言
---------------------------------------------
vi /etc/sysconfig/i18n
LANG="en_US.UTF-8"
// en_US.UTF-8 ： 英文  zh_CN.UTF-8 ：简体中文
// 如果需要设置在中文状态下不出乱码可以设置为： LANG="zh_CN.GB18030"



---------------------------------------------
三、设置网卡
---------------------------------------------
vim /etc/sysconfig/network-scripts/eth0
************************
DEVICE=eth0
HWADDR=08:00:27:CA:BD:24
TYPE=Ethernet
IPADDR=192.168.15.178
NETMASK=255.255.255.0
NETWORK=192.168.15.1
ONBOOT=yes
NM_CONTROLLED=no
BOOTPROTO=none
*********************
